if (linhaEncontrada !== -1) { sheet.getRange(linhaEncontrada, 1, 1, linhaDados.length).setValues([linhaDados]); } else { sheet.appendRow(linhaDados); } return gerarRespostaJson({"status": "sucesso", "pdf": urlPdf}); } catch (err) { return gerarRespostaJson({"status": "erro", "msg": err.message}); } } // --- MANTENHA AS FUNÇÕES criarPdfOS, salvarNoDrive e gerarRespostaJson ABAIXO IGUAIS --- function gerarRespostaJson(objeto) { return ContentService.createTextOutput(JSON.stringify(objeto)).setMimeType(ContentService.MimeType.JSON); } function criarPdfOS(dados, os) { const pastaPDF = DriveApp.getFolderById(FOLDER_PDF_ID); const nomeArquivoPDF = "OS_" + os + ".pdf"; const arquivosExistentes = pastaPDF.getFilesByName(nomeArquivoPDF); while (arquivosExistentes.hasNext()) { arquivosExistentes.next().setTrashed(true); } const html = `

Cliente: ${dados.nome || "Não informado"}

Aparelho: ${dados.aparelho || "Não informado"}

Valor: ${dados.valor || "R$ 0,00"}

${dados.assinatura ? `
` : ""}
`; const blob = Utilities.newBlob(html, "text/html", "temp.html"); const pdf = pastaPDF.createFile(blob.getAs("application/pdf")).setName(nomeArquivoPDF); pdf.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return pdf.getUrl(); } function salvarNoDrive(base64, caminhoCompleto, folderId) { if (!base64 || base64.length < 10) return; const nomeArquivo = caminhoCompleto.split('/').pop(); const folder = DriveApp.getFolderById(folderId); const arquivosExistentes = folder.getFilesByName(nomeArquivo); while (arquivosExistentes.hasNext()) { arquivosExistentes.next().setTrashed(true); } const decoded = Utilities.base64Decode(base64.split(",")[1]); folder.createFile(Utilities.newBlob(decoded, "image/png", nomeArquivo)); }
